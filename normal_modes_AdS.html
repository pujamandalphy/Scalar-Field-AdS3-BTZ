import numpy as np
from scipy.integrate import solve_ivp
from scipy.optimize import bisect

L = 1.0
m2 = 0.0
ell = 0
eps = 1e-6
rho_max = 10.0

Delta = 1 + np.sqrt(1 + m2)

def radial_system(rho, y, omega):
    R, Rp = y
    s = np.sinh(rho)
    c = np.cosh(rho)
    Rpp = (
        -(c**2 + s**2)/(s*c) * Rp
        + (ell**2/s**2 - omega**2/c**2 - m2) * R
    )
    return [Rp, Rpp]

def initial_conditions():
    return [eps**abs(ell), 0.0]

def shoot(omega):
    sol = solve_ivp(
        lambda r, y: radial_system(r, y, omega),
        [eps, rho_max],
        initial_conditions(),
        rtol=1e-8, atol=1e-8
    )
    return sol.y[0, -1]

w0 = bisect(shoot, 1.5, 2.5)
print("First eigenfrequency:", w0)
